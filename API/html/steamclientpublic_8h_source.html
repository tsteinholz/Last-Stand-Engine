816 inline bool CSteamID::IsValid() const
817 {
818  if ( m_steamid.m_comp.m_EAccountType <= k_EAccountTypeInvalid || m_steamid.m_comp.m_EAccountType >= k_EAccountTypeMax )
819  return false;
820 
821  if ( m_steamid.m_comp.m_EUniverse <= k_EUniverseInvalid || m_steamid.m_comp.m_EUniverse >= k_EUniverseMax )
822  return false;
823 
824  if ( m_steamid.m_comp.m_EAccountType == k_EAccountTypeIndividual )
825  {
826  if ( m_steamid.m_comp.m_unAccountID == 0 || m_steamid.m_comp.m_unAccountInstance > k_unSteamUserWebInstance )
827  return false;
828  }
829 
830  if ( m_steamid.m_comp.m_EAccountType == k_EAccountTypeClan )
831  {
832  if ( m_steamid.m_comp.m_unAccountID == 0 || m_steamid.m_comp.m_unAccountInstance != 0 )
833  return false;
834  }
835 
836  if ( m_steamid.m_comp.m_EAccountType == k_EAccountTypeGameServer )
837  {
838  if ( m_steamid.m_comp.m_unAccountID == 0 )
839  return false;
840  // Any limit on instances? We use them for local users and bots
841  }
842  return true;
843 }
