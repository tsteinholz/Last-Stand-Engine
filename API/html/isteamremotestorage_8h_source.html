145 enum EUGCReadAction
146 {
147  // Keeps the file handle open unless the last byte is read. You can use this when reading large files (over 100MB) in sequential chunks.
148  // If the last byte is read, this will behave the same as k_EUGCRead_Close. Otherwise, it behaves the same as k_EUGCRead_ContinueReading.
149  // This value maintains the same behavior as before the EUGCReadAction parameter was introduced.
150  k_EUGCRead_ContinueReadingUntilFinished = 0,
151 
152  // Keeps the file handle open. Use this when using UGCRead to seek to different parts of the file.
153  // When you are done seeking around the file, make a final call with k_EUGCRead_Close to close it.
154  k_EUGCRead_ContinueReading = 1,
155 
156  // Frees the file handle. Use this when you're done reading the content.
157  // To read the file from Steam again you will need to call UGCDownload again.
