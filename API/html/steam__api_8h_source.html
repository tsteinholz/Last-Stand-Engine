262 //-----------------------------------------------------------------------------
263 // Purpose: maps a steam callback to a class member function
264 // template params: T = local class, P = parameter struct
265 //-----------------------------------------------------------------------------
266 template< class T, class P, bool bGameServer >
267 class CCallback : protected CCallbackBase
268 {
269 public:
270  typedef void (T::*func_t)( P* );
271 
272  // If you can't support constructing a callback with the correct parameters
273  // then uncomment the empty constructor below and manually call
274  // ::Register() for your object
275  // Or, just call the regular constructor with (NULL, NULL)
276  // CCallback() {}
277 
278  // constructor for initializing this object in owner's constructor
279  CCallback( T *pObj, func_t func ) : m_pObj( pObj ), m_Func( func )
280  {
281  if ( pObj && func )
282  Register( pObj, func );
283  }
284 
285  ~CCallback()
286  {
287  if ( m_nCallbackFlags & k_ECallbackFlagsRegistered )
288  Unregister();
289  }
290 
291  // manual registration of the callback
292  void Register( T *pObj, func_t func )
293  {
294  if ( !pObj || !func )
295  return;
296 
297  if ( m_nCallbackFlags & k_ECallbackFlagsRegistered )
298  Unregister();
299 
300  if ( bGameServer )
301  {
302  m_nCallbackFlags |= k_ECallbackFlagsGameServer;
303  }
304  m_pObj = pObj;
305  m_Func = func;
306  // SteamAPI_RegisterCallback sets k_ECallbackFlagsRegistered
307  SteamAPI_RegisterCallback( this, P::k_iCallback );
308  }
309 
310  void Unregister()
311  {
312  // SteamAPI_UnregisterCallback removes k_ECallbackFlagsRegistered
313  SteamAPI_UnregisterCallback( this );
314  }
315 
316  void SetGameserverFlag() { m_nCallbackFlags |= k_ECallbackFlagsGameServer; }
317 protected:
318  virtual void Run( void *pvParam )
319  {
320  (m_pObj->*m_Func)( (P *)pvParam );
321  }
322  virtual void Run( void *pvParam, bool, SteamAPICall_t )
323  {
324  (m_pObj->*m_Func)( (P *)pvParam );
325  }
326  int GetCallbackSizeBytes()
327  {
328  return sizeof( P );
329  }
330 
331  T *m_pObj;
332  func_t m_Func;
333 };
